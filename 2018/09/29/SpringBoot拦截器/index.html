import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.web.servlet.HandlerInterceptor;
import org.springframework.web.servlet.config.annotation.InterceptorRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurerAdapter;

@Configuration
public class MyWebAppConfigurer extends WebMvcConfigurerAdapter {

@Bean
public HandlerInterceptor getMyInterceptor(){
    return new LoginInterceptor();
}

@Override
public void addInterceptors(InterceptorRegistry registry) {
    // addPathPatterns 用于添加拦截规则, 这里假设拦截 /dcairAdmin 后面的全部链接

    registry.addInterceptor(getMyInterceptor()).addPathPatterns("/dcairAdmin/**");
    super.addInterceptors(registry);
}

}

package com.dcair.admin.manage.configuration;

import com.dcair.admin.manage.constant.Constant;
import com.dcair.admin.manage.entity.User;
import com.dcair.admin.manage.entity.vo.DataVO;
import com.dcair.admin.manage.ocnfig.redis.RedisClient;
import com.dcair.admin.manage.utils.GsonUtil;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.servlet.HandlerInterceptor;
import org.springframework.web.servlet.ModelAndView;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.servlet.http.HttpSession;
import java.io.IOException;
import java.io.OutputStreamWriter;
import java.io.PrintWriter;
import java.util.HashMap;
import java.util.Map;

public class LoginInterceptor implements HandlerInterceptor {

@Autowired
private RedisClient redisClient;

// 在执行handler返回modelAndView之前执行
@Override
public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {

    //得到请求的url
    String url=request.getRequestURI();
    System.out.println(url);
    String[] strs=url.split("/");

    //判断是否是公开地址
    //实际开发中需要公开地址配置在配置文件中
    if(strs[2].equals("public")||strs[3].equals("login")) {
        //如果是公开地址则放行
        return true;
    }

    //判断用户是否登录
    HttpSession session=request.getSession();
    User user=(User) session.getAttribute("loginer");
    if(user!=null){
        Boolean isLogin=redisClient.exists(user.getUserName()+Constant.redisLoginKeySuffix);
        if(isLogin==true){
            //保存登录状态key的生存时间 (15分钟)
            redisClient.set(user.getUserName()+Constant.redisLoginKeySuffix,"1",900l );
            return true;
        }
        else{
            threwLoginOutException(response);
            return true;
        }
    }

    else {
        threwLoginOutException(response);
        return true;
    }


}


private void threwLoginOutException(HttpServletResponse response) throws IOException{
    DataVO dataVO=new DataVO();
    PrintWriter writer=null;
    OutputStreamWriter output=null;
    Map<String,String> error = new HashMap<String,String>();
    error.put("msg","用户未登录或登录失效");
    error.put("code","401");
    dataVO.setError(error);
    dataVO.setSuccess(false);

    output=new OutputStreamWriter(response.getOutputStream(),"UTF-8");
    writer=new PrintWriter(output,true);
    String jsonStr=GsonUtil.obj2Json(dataVO);
    writer.write(jsonStr);
    writer.flush();
    writer.close();
    output.close();
}

// 在执行handler返回modelAndView之前执行
@Override
public void postHandle(HttpServletRequest request, HttpServletResponse response, Object handler, ModelAndView modelAndView) throws Exception {

}

// 在执行handler之后执行的
@Override
public void afterCompletion(HttpServletRequest request, HttpServletResponse response, Object handler, Exception exception) throws Exception {

}

}

package com.dcair.admin.manage.utils;

import com.google.gson.Gson;

/**

  • @Description Gson工具
  • @Author guanjiagui

  • @CreateDate：2018年2月6日 下午2:55:10
    */
    public class GsonUtil {
    private static Gson gson = new Gson();

    /**

    • obj to json
    • @param obj
    • @return String

    • @since project1.0
      */
      public static String obj2Json(Object obj) {
      return gson.toJson(obj);
      }

      /**

    • json to obj
    • @param json
    • @param classOfT
    • @return
    • @since project1.0
      */
      public static T json2Obj(String json,Class classOfT) {
      return gson.fromJson(json, classOfT);
      }

}
