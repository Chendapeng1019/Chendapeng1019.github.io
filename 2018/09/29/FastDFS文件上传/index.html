FastDFSClient实体类：

public class FastDFSClient {

private StorageClient storageClient;

private static FastDFSClient fastDfsClient;

/**
 * 创建文件上传客户端
 * @return
 * @throws FileNotFoundException
 */
public static FastDFSClient build() throws FileNotFoundException {
    if (fastDfsClient == null) {
        fastDfsClient = new FastDFSClient();
    }
    return fastDfsClient;
}


private FastDFSClient() throws FileNotFoundException {
    String fdfsFilePath = null;
    try {
        // 服务器绝对路径

// fdfsFilePath = “/usr/local/java/properties/fastdfs_conf.properties”;
// macOS下的绝对路径
// fdfsFilePath = “/Users/lvshitao/ideaProject/CGA_MicroWebsite_Admin-013/src/main/resource/fastdfs_conf.properties”;
// windows下的绝对路径
//fdfsFilePath = “E:\ideaProject\CGA_MicroWebsite_Admin-013\src\main\resource\fastdfs_conf.properties”;
//fdfsFilePath =”F:\vue\CGA_MicroWebsite_AdminFront-013\src\resource\fastdfs_conf.properties”;

        fdfsFilePath ="F:\\fastdfs_conf.properties";
    } catch (Exception e) {
        throw new FileNotFoundException("fdfs_client.properties");
    }
    init(fdfsFilePath);
}

/**
 * 初始化文件上传客户端
 * @param fdfsFilePath
 */
private void init(String fdfsFilePath) {
    try {
        LogUtil.record("初始化文件上传客户端");
        ClientGlobal.init(fdfsFilePath);
        LogUtil.record("trackerClient:"+ new TrackerClient());
        TrackerClient trackerClient = new TrackerClient();
        LogUtil.record("trackerServer:"+ trackerClient.getConnection());
        TrackerServer trackerServer = trackerClient.getConnection();
        LogUtil.record("storageServer:"+ trackerClient.getStoreStorage(trackerServer));
        StorageServer storageServer = null;
        storageClient = new StorageClient(trackerServer, storageServer);
    } catch (Exception e) {
        e.printStackTrace();
        throw new RuntimeException("init FastDfs error" + e.getMessage());
    }
}

/**
 * 文件上传
 * @param fileBytes
 * @param fileName
 * @return
 * @throws IOException
 * @throws MyException
 */
public String uploadFile(byte[] fileBytes, String fileName)
        throws IOException, MyException {
    String[] ss = storageClient.upload_file(fileBytes, fileName, null);
    StringBuffer sbPath = new StringBuffer();
    for (String string : ss) {
        sbPath.append("/" + string);
    }
    return sbPath.toString();
}

/**
 * 文件上传
 * @param filePath

 */
public String uploadFile(String filePath, String extName)
        throws IOException, MyException {
    String[] ss = storageClient.upload_file(filePath, extName, null);
    StringBuffer sbPath = new StringBuffer();
    for (String string : ss) {
        sbPath.append("/" + string);

}


/**
 * 文件上传
 * @param filePath

 * @return
 * @throws IOException

 */
public String uploadFileByte(byte [] filbyts, String extName)
        throws IOException, MyException {
    String[] ss = storageClient.upload_file(filbyts, extName, null);
    StringBuffer sbPath = new StringBuffer();
    for (String string : ss) {
        sbPath.append("/" + string);
    }
    return sbPath.toString();
}

/**
 * 文件删除
 * @param filePath
 * @return
 * @throws FileNotFoundException
 * @throws IOException
 * @throws MyException
 */
public int delFile(String filePath) throws FileNotFoundException,
        IOException, MyException {
    String group = filePath.split("/")[0];
    String path = filePath.replaceFirst("", group);
    int i = storageClient.delete_file(group, path);
    return i;
}
